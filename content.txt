  $(document).ready(function () {

      setupAvatar();
    }

    sfIndex.individualResize = function () {
      this.scaleContentContainers(1024, 688, 80, 9); //indices.js
      setupAvatarHeight();
    }
    sfIndex.start();
  });

  function setupAvatar() {
    var cookiesLocal = Cookies.getLocalCookies();
    var xterCookie =  cookiesLocal['h5'] || '0251213045000';

    $.get("/h/me/me-xter/me-threequarter-front-optimize.php?h5="+xterCookie, function(data, status) {
      $('.avatar').attr('src','data:image/png;base64,'+data);
      setupAvatarHeight();
    });
  }

  function setupAvatarHeight() {
    var h = $('.content-container').height() * 0.23;
    $('.avatar-img-container').height(h + 'px');
  }
